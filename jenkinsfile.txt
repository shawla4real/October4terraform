pipeline{
    agent any 
    tools {
        terraform 'terraform2'
    }
    stages{
        stage('Git checkout'){
            steps{
            git 'https://github.com/shawla4real/terraform10-4.git' 
         }
    }     
        stage('Terraform Init'){
            steps{
            sh 'terraform init' 
         }   
        }
        stage('Terraform apply'){
            steps{
            sh 'terraform init --auto-approve'         
         }
       }
    }
}
